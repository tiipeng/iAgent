# openai 1.32+ adds a hard dependency on `jiter`, a Rust-compiled JSON parser
# with no prebuilt wheels for iOS (machine reports as iPad*,* under darwin).
# Pin to the last pre-jiter release so pip never needs Rust to install.
openai>=1.30,<1.32

# pydantic v2 has a `pydantic-core` Rust crate — also fails to build on iOS
# because puccinialin does not recognise the iPad target triple. openai 1.31
# still supports pydantic v1 via a compatibility shim, and v1 is pure Python.
pydantic>=1.10,<2

# python-telegram-bot, aiofiles, aiosqlite, python-dotenv, PyYAML are pure Python.
python-telegram-bot[job-queue]>=20.7,<22

# aiohttp ships C extensions; its sdist build needs no Rust, just a C compiler.
# Procursus provides clang, so source builds succeed if a wheel does not match.
aiohttp>=3.9,<3.10
aiofiles>=23.2
aiosqlite>=0.20
python-dotenv>=1.0
# PyYAML 6.0.3 stopped publishing the cp39 universal2 wheel and pip falls back
# to the source dist, which needs a C compiler (Procursus does not ship one
# by default). 6.0.2 still has the prebuilt wheel.
PyYAML>=6.0,<6.0.3
